	prototype: "montage/ui/css-declaration-component",
	properties: {
		element:"$.ui-backgound-color{backgound-color}"
	},

	}

	prototype: "montage/ui/css-declaration-component",
	properties: {
		element:"$.ui-backgound-color"
	},
	bindings: {
		"background-color":"<-owner.contentColor", "converter": {"@": "darkenConverter"};
	}
},
